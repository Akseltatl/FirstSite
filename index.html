import pygame # Размеры окна SCREEN_WIDTH = 800 SCREEN_HEIGHT = 600 # Цвета WHITE = (255, 255, 255) BLACK = (0, 0, 0) GRAY = (100, 100, 100) # Физические параметры MAX_SPEED = 10 ACCELERATION = 0.2 DECELERATION = 0.1 TURN_SPEED = 3 # градусов за кадр # Пути к ресурсам CAR_IMAGES = [ 'assets/cars/car_blue.png', 'assets/cars/car_red.png', 'assets/cars/car_green.png' ] BACKGROUND_IMAGE = 'assets/background/road.png' import pygame import sys import random from config import * def load_and_scale_image(path, scale=0.2): """Загружает и масштабирует изображение.""" img = pygame.image.load(path) width, height = img.get_size() new_size = (int(width * scale), int(height * scale)) return pygame.transform.scale(img, new_size) def main(): pygame.init() screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT)) pygame.display.set_caption("Car Driving Game") clock = pygame.time.Clock() # Загружаем ресурсы background = pygame.image.load(BACKGROUND_IMAGE).convert() car_images = [load_and_scale_image(path) for path in CAR_IMAGES] font = pygame.font.Font(None, 36) # Выбираем случайную машину car_img = random.choice(car_images) car_rect = car_img.get_rect(center=(SCREEN_WIDTH // 2, SCREEN_HEIGHT // 2)) # Параметры движения speed = 0 angle = 0 # угол поворота машины running = True while running: # Обработка событий for event in pygame.event.get(): if event.type == pygame.QUIT: running = False # Управление keys = pygame.key.get_pressed() if keys[pygame.K_UP]: speed += ACCELERATION if keys[pygame.K_DOWN]: speed -= DECELERATION if keys[pygame.K_LEFT]: angle += TURN_SPEED if keys[pygame.K_RIGHT]: angle -= TURN_SPEED # Ограничиваем скорость speed = max(-MAX_SPEED / 2, min(speed, MAX_SPEED)) # Двигаем машину radians = pygame.math.Vector2(0, -1).rotate(angle).as_polar()[1] car_rect.x += speed * pygame.math.cos(radians) car_rect.y += speed * pygame.math.sin(radians) # Прокрутка фона (имитация движения) rel_y = car_rect.y % background.get_rect().height screen.blit(background, (0, rel_y - background.get_rect().height)) if rel_y < SCREEN_HEIGHT: screen.blit(background, (0, rel_y)) # Рисуем машину rotated_img = pygame.transform.rotate(car_img, angle) rotated_rect = rotated_img.get_rect(center=car_rect.center) screen.blit(rotated_img, rotated_rect) # Отображаем скорость speed_text = font.render(f"Speed: {abs(speed):.1f}", True, WHITE) screen.blit(speed_text, (10, 10)) pygame.display.flip() clock.tick(60) # 60 FPS pygame.quit() sys.exit() if __name__ == "__main__": main()
